\newcommand{\expEntry}[5]{%
  \vspace{2pt}%
  \ifthenelse{\equal{#1}{}}{}{%
    \textbf{#1} \hfill \textbf{#2} \\[1pt]%
  }%
  \textit{#3} \hfill \textit{#4}%
  \ifthenelse{\equal{#5}{}}{}{%
    \\[1pt]\small\textit{#5}%
  }%
  \vspace{0pt}%
}

\newcommand{\eduEntry}[4]{%
  \vspace{2pt}%
  \textbf{#1} \hfill \textbf{#2} \\[1pt]%
  \textit{#3} \hfill \textit{#4}%
  \vspace{0pt}%
}
